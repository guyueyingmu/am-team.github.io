        ap.getStorage(OPTIONS | key, CALLBACK)

        获取缓存数据。可直接传一个字符串，作为 OPTIONS.key。

        OPTIONS 参数说明

        名称类型必填描述
        keyString是缓存数据的 key

        CALLBACK 参数说明

        名称 | 类型 | 描述 data | Object | 相应 key 的数据。若未找到数据会返回 null

        代码演示

        <script src="https://a.alipayobjects.com/g/h5-lib/alipayjsapi/0.2.4/alipayjsapi.inc.min.js"></script>

<button id="J_btn_set" class="btn btn-default">写入缓存数据</button>
<button id="J_btn_get" class="btn btn-default">获取缓存数据</button>
<script>
  var btnSet= document.querySelector('#J_btn_set');
  var btnGet= document.querySelector('#J_btn_get');
  btnSet.addEventListener('click', function(){
    ap.setStorage({
      key: 'currentCity',
      data: {
        cityName: '杭州',
        adCode: '330100',
        spell: ' hangzhou'
      },
      success: function() {
        ap.alert('写入成功');
      }
    });
  });
  btnGet.addEventListener('click', function(){
    ap.getStorage({
      key: 'currentCity',
      success: function(res) {
        ap.alert('获取成功：' + res.data.cityName);
      },
      fail: function(res){
        ap.alert(res.errorMessage);
      }
    });
  });
</script>
